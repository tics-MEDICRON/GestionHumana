CREATE TABLE IF NOT EXISTS `evaluacion_desempeno` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `idPersona` varchar(15) NOT NULL,
  `fechaInicio` date NOT NULL,
  `fechaFin` date NOT NULL,
  `estado` varchar(20) NOT NULL DEFAULT 'Abierta',
  `resultadoDesempeno` double DEFAULT 0,
  `resultadoCompetencia` double DEFAULT 0,
  `resultadoFinal` double DEFAULT 0,
  `created_at` datetime NOT NULL DEFAULT current_timestamp(),
  PRIMARY KEY (`id`),
  KEY `idx_evaluacion_desempeno_persona` (`idPersona`),
  CONSTRAINT `fk_evaluacion_desempeno_persona` FOREIGN KEY (`idPersona`) REFERENCES `persona` (`identificacion`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;

ALTER TABLE `desempeno`
  ADD COLUMN `idEvaluacionDesempeno` int(11) DEFAULT NULL,
  ADD KEY `idx_desempeno_evaluacion` (`idEvaluacionDesempeno`),
  ADD CONSTRAINT `fk_desempeno_evaluacion` FOREIGN KEY (`idEvaluacionDesempeno`) REFERENCES `evaluacion_desempeno` (`id`);

ALTER TABLE `evaluacioncompetencia`
  ADD COLUMN `idEvaluacionDesempeno` int(11) DEFAULT NULL,
  ADD KEY `idx_evaluacioncompetencia_evaluacion` (`idEvaluacionDesempeno`),
  ADD CONSTRAINT `fk_evaluacioncompetencia_evaluacion` FOREIGN KEY (`idEvaluacionDesempeno`) REFERENCES `evaluacion_desempeno` (`id`);
